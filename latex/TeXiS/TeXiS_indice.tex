% Nombre Indice Palabras
\def\indicePalabrasVal{Index}
\newcommand{\indicePalabras}[1]{
\def\indicePalabrasVal{#1}
}

% Nombre Indice Citas
\def\indiceCitasVal{Index}
\newcommand{\indiceCitas}[1]{
\def\indiceCitasVal{#1}
}

% Para pedir que se genere  la información para el índice de palabras.
% Para  no  ralentizar la  compilación,  sólo  se  genera cuando  está
% definido el símbolo correspondiente.

\ifx\generaindice\undefined
\else
\makeindex
\fi

% Pruebas con índices
\usepackage{index}
\newindex{default}{idx}{ind}{\indicePalabrasVal}
%\newindex{default}{idx}{ind}{Índice}
\newindex{cite}{cdx}{cnd}{\indiceCitasVal}
%\newindex{cite}{cdx}{cnd}{Índice de citas}
\renewcommand{\citeindextype}{cite}
\citeindextrue
